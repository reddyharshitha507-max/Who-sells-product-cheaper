const CATS=['All',...new Set(PRODUCTS.map(p=>p.cat))];
let activeCat='All';
const CAT_ICONS={'All':'🌐','Cricket Kit':'🏏','Beauty':'💄','Electronics':'📱',"Women's Fashion":'👗',"Men's Fashion":'👔',"Kids' Fashion":'👶','Footwear':'👟','Daily Essentials':'🧴'};

const fmt=n=>'₹'+n.toLocaleString('en-IN');
const disc=(p,o)=>(o&&o>p)?Math.round((1-p/o)*100):null;
function cheapKey(prices){let min=Infinity,k=null;Object.entries(prices).forEach(([key,v])=>{if(v.p&&v.p<min){min=v.p;k=key;}});return k;}
function maxSave(prices){const v=Object.values(prices).filter(x=>x.p).map(x=>x.p);return v.length>1?Math.max(...v)-Math.min(...v):0;}
function minPr(prices){const v=Object.values(prices).filter(x=>x.p).map(x=>x.p);return v.length?Math.min(...v):Infinity;}
function maxPr(prices){const v=Object.values(prices).filter(x=>x.p).map(x=>x.p);return v.length?Math.max(...v):0;}
function hl(text,q){if(!q)return text;const re=new RegExp('('+q.replace(/[.*+?^${}()|[\]\\]/g,'\\$&')+')','gi');return text.replace(re,'<mark>$1</mark>');}

// Build tabs
(function(){
  const wrap=document.getElementById('catTabs');
  CATS.forEach(c=>{
    const b=document.createElement('button');
    b.className='ctab'+(c==='All'?' active':'');
    b.textContent=(CAT_ICONS[c]||'📦')+' '+c;
    b.onclick=()=>{activeCat=c;document.querySelectorAll('.ctab').forEach(t=>t.classList.remove('active'));b.classList.add('active');renderAll();};
    wrap.appendChild(b);
  });
})();

function handleSearch(){
  const v=document.getElementById('searchInput').value;
  document.getElementById('clearBtn').style.display=v?'block':'none';
  renderAll();
}
function clearSearch(){
  document.getElementById('searchInput').value='';
  document.getElementById('clearBtn').style.display='none';
  renderAll();
}

function renderAll(){
  const q=document.getElementById('searchInput').value.toLowerCase().trim();
  const sort=document.getElementById('sortSel').value;

  let list=PRODUCTS.filter(p=>{
    const mQ=!q||p.name.toLowerCase().includes(q)||p.cat.toLowerCase().includes(q)||(p.tags&&p.tags.toLowerCase().includes(q));
    const mC=activeCat==='All'||p.cat===activeCat;
    return mQ&&mC;
  });

  if(sort==='savings')list.sort((a,b)=>maxSave(b.prices)-maxSave(a.prices));
  else if(sort==='low')list.sort((a,b)=>minPr(a.prices)-minPr(b.prices));
  else if(sort==='high')list.sort((a,b)=>maxPr(b.prices)-maxPr(a.prices));
  else if(sort==='az')list.sort((a,b)=>a.name.localeCompare(b.name));

  const grid=document.getElementById('grid');
  const noRes=document.getElementById('noRes');
  const rCount=document.getElementById('resultCount');
  grid.innerHTML='';

  if(!list.length){
    noRes.style.display='block';
    document.getElementById('noResQ').textContent=q||activeCat;
    rCount.textContent='';return;
  }
  noRes.style.display='none';
  rCount.textContent=q?`${list.length} result${list.length!==1?'s':''} for "${q}"`:`Showing ${list.length} product${list.length!==1?'s':''}`;

  list.forEach((prod,idx)=>{
    const bk=cheapKey(prod.prices);
    const bs=STORES.find(s=>s.key===bk);
    const sv=maxSave(prod.prices);
    const allP=Object.values(prod.prices).filter(v=>v.p).map(v=>v.p);
    const mx=allP.length?Math.max(...allP):1;
    const sp=mx>0?Math.round(sv/mx*100):0;

    const card=document.createElement('div');
    card.className='pcard';
    card.style.animationDelay=`${Math.min(idx,12)*0.045}s`;

    card.innerHTML=`
      <div class="chead">
        <div class="pemoji">${prod.emoji}</div>
        <div class="pmeta">
          <div class="pname">${hl(prod.name,q)}</div>
          <div class="pcat-badge">${CAT_ICONS[prod.cat]||''} ${prod.cat}</div>
        </div>
        ${bs?`<div class="pbest"><div class="pbest-lbl">Best price</div><div class="pbest-val">${bs.icon} ${bs.name}</div></div>`:''}
      </div>`;

    const row=document.createElement('div');
    row.className='prow';

    STORES.forEach(st=>{
      const d=prod.prices[st.key];
      const isBest=st.key===bk;
      const cell=document.createElement('div');
      cell.className=`pcell ${st.tc}${isBest?' best':''}`;

      if(!d||!d.p){
        // Not available — still show a visit button
        cell.innerHTML=`
          <div class="csname">${st.name}</div>
          <div class="cna">Not on ${st.name}</div>
          <button class="shop-btn na-btn" onclick="window.open('${st.searchUrl(prod.name)}','_blank')">↗ Browse ${st.name}</button>
        `;
      } else {
        const pct=disc(d.p,d.o);
        // Build button — clicking searches this product on that store
        const searchURL=st.searchUrl(prod.name);
        cell.innerHTML=`
          <div class="csname">${st.name}</div>
          <div class="cprice">${fmt(d.p)}</div>
          ${d.o?`<div class="corig">${fmt(d.o)}</div>`:''}
          ${pct?`<div class="coff">-${pct}% OFF</div>`:''}
          <button class="shop-btn" onclick="window.open('${searchURL}','_blank')">${st.icon} Shop on ${st.name}</button>
        `;
      }
      row.appendChild(cell);
    });

    card.appendChild(row);

    const foot=document.createElement('div');
    foot.className='cfoot';
    foot.innerHTML=`<span class="fl">Save up to</span><span class="fa">${fmt(sv)} (${sp}%)</span><div class="bwrap"><div class="bfill" style="width:${sp}%"></div></div><span class="fl">by choosing wisely 💡</span>`;
    card.appendChild(foot);
    grid.appendChild(card);
  });
}

renderAll();
